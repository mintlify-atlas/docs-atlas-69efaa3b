---
title: 'List Categories'
api: 'GET /api/categories'
description: 'Retrieve a paginated list of all categories'
---

## Endpoint

```
GET /api/categories
```

## Authentication

No authentication required. This is a public endpoint.

## Query Parameters

<ParamField query="page" type="integer" default="0">
  Page number (0-indexed)
</ParamField>

<ParamField query="size" type="integer" default="20">
  Number of items per page
</ParamField>

<ParamField query="sort" type="string" default="id">
  Sort field and direction (e.g., "id", "name,asc", "name,desc")
</ParamField>

## Response

<ResponseField name="success" type="boolean" required>
  Indicates if the request was successful
</ResponseField>

<ResponseField name="timestamp" type="string" required>
  ISO 8601 timestamp of the response
</ResponseField>

<ResponseField name="data" type="object" required>
  Paginated category data
  
  <ResponseField name="content" type="array" required>
    Array of category objects
    
    <ResponseField name="id" type="long" required>
      Unique identifier for the category
    </ResponseField>
    
    <ResponseField name="name" type="string" required>
      Name of the category
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="pageable" type="object" required>
    Pagination information
    
    <ResponseField name="pageNumber" type="integer">
      Current page number (0-indexed)
    </ResponseField>
    
    <ResponseField name="pageSize" type="integer">
      Number of items per page
    </ResponseField>
    
    <ResponseField name="offset" type="integer">
      Offset from the start of the dataset
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="totalElements" type="long" required>
    Total number of categories across all pages
  </ResponseField>
  
  <ResponseField name="totalPages" type="integer" required>
    Total number of pages
  </ResponseField>
  
  <ResponseField name="size" type="integer" required>
    Number of items per page
  </ResponseField>
  
  <ResponseField name="number" type="integer" required>
    Current page number (0-indexed)
  </ResponseField>
  
  <ResponseField name="first" type="boolean" required>
    Whether this is the first page
  </ResponseField>
  
  <ResponseField name="last" type="boolean" required>
    Whether this is the last page
  </ResponseField>
  
  <ResponseField name="empty" type="boolean" required>
    Whether the page is empty
  </ResponseField>
</ResponseField>

## Example Request

```bash
curl -X GET "http://localhost:8080/api/categories?page=0&size=10&sort=name,asc"
```
