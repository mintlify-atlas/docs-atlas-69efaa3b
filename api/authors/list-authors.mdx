---
title: 'List Authors'
api: 'GET /api/authors'
description: 'Retrieve a paginated list of all authors'
---

## Endpoint

```
GET /api/authors
```

## Authentication

Requires authentication with a valid Bearer token.

## Query Parameters

<ParamField query="page" type="integer" default="0">
  Page number (zero-indexed)
</ParamField>

<ParamField query="size" type="integer" default="20">
  Number of items per page
</ParamField>

<ParamField query="sort" type="string" default="id">
  Sort field and direction (e.g., `id,asc` or `name,desc`)
</ParamField>

## Response

<ResponseField name="success" type="boolean" required>
  Indicates if the request was successful
</ResponseField>

<ResponseField name="timestamp" type="string" required>
  ISO 8601 timestamp of the response
</ResponseField>

<ResponseField name="data" type="object" required>
  Paginated author data
  
  <ResponseField name="content" type="array" required>
    Array of author objects
    
    <ResponseField name="id" type="long" required>
      Unique identifier for the author
    </ResponseField>
    
    <ResponseField name="name" type="string" required>
      Full name of the author
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="pageable" type="object" required>
    Pagination information
    
    <ResponseField name="pageNumber" type="integer" required>
      Current page number (zero-indexed)
    </ResponseField>
    
    <ResponseField name="pageSize" type="integer" required>
      Number of items per page
    </ResponseField>
    
    <ResponseField name="offset" type="long" required>
      Offset of the first item in the current page
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="totalPages" type="integer" required>
    Total number of pages
  </ResponseField>
  
  <ResponseField name="totalElements" type="long" required>
    Total number of authors
  </ResponseField>
  
  <ResponseField name="last" type="boolean" required>
    Whether this is the last page
  </ResponseField>
  
  <ResponseField name="first" type="boolean" required>
    Whether this is the first page
  </ResponseField>
  
  <ResponseField name="numberOfElements" type="integer" required>
    Number of elements in the current page
  </ResponseField>
  
  <ResponseField name="empty" type="boolean" required>
    Whether the page is empty
  </ResponseField>
</ResponseField>

## Example Request

```bash
curl -X GET "http://localhost:8080/api/authors?page=0&size=10&sort=name,asc" \
  -H "Authorization: Bearer YOUR_ACCESS_TOKEN" \
  -H "Content-Type: application/json"
```
